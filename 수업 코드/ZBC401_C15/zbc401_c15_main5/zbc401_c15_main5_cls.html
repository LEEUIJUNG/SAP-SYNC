  METHOD CLASS_CONSTRUCTOR.
    SELECT * FROM SAPLANE INTO TABLE GT_PLANETYPES.

  METHOD GET_TECHNICAL_ATTRIBUTES.
    DATA: LS_PLANETYPE LIKE LINE OF GT_PLANETYPES.

    READ TABLE GT_PLANETYPES INTO LS_PLANETYPE
    WITH KEY PLANETYPE = IV_PLANETYPE.

    CASE SY-SUBRC.
      WHEN 0.
        EV_WEIGHT = LS_PLANETYPE-WEIGHT.
        EV_WEI_UNIT = LS_PLANETYPE-WEI_UNIT.
        EV_TANKCAP = LS_PLANETYPE-TANKCAP.
        EV_CAP_UNIT = LS_PLANETYPE-CAP_UNIT.
      WHEN OTHERS.
        RAISE WRONG_PLANETYPE.
    ENDCASE.
  ENDMETHOD.
