MODULE EXIT_0100 INPUT.
  CASE OK_CODE.
    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN 'CANC'.
      " 검색조건 초기화, STATUS ICON 초기화
     CLEAR : SDYN_BOOK.

      CLEAR: SBOOK, SCUSTOM, STRAVELAG, SPFLI, SFLIGHT.

      CLEAR: SCR_0110, SCR_0120.

      GV_STATUS_ICON = GC_INIT.

      " 화면 새로고침
      LEAVE SCREEN.

    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

MODULE CHECK_BOOKING INPUT.

  CLEAR: SBOOK,
  SCUSTOM,
  STRAVELAG,
  SCR_0110.

* 입력필드 값 기준으로 예약정보 값 1건을 조회
  SELECT SINGLE *
    FROM SBOOK
    WHERE CARRID = SDYN_BOOK-CARRID
    AND CONNID = SDYN_BOOK-CONNID
    AND FLDATE = SDYN_BOOK-FLDATE
    AND BOOKID = SDYN_BOOK-BOOKID.

  IF SY-SUBRC = 0.
    GV_SELECT_RESULT = GC_SELECT_SUCCESS.
*----------------------------------------------------------------------*
    " 0110 화면을 위한 데이터 조회
    PERFORM SELECT_SINGLE_CUSTOMER.
    PERFORM SELECT_SINGLE_TRAVELAG.

    PERFORM SELECT_FIX_VALUE_CUSTTYPE CHANGING SCR_0110-CUSTTYPE_TEXT.
    PERFORM SELECT_FIX_VALUE_CLASS CHANGING SCR_0110-CLASS_TEXT.
*----------------------------------------------------------------------*

*----------------------------------------------------------------------*
    " 0120 화면을 위한 데이터 조회
    PERFORM SELECT_SINGLE_CONNECTION.
    PERFORM SELECT_SINGLE_FLIGHT.

* 출발국가명을 가져온다.
    PERFORM SELECT_COUNTRY_NAME USING SPFLI-COUNTRYFR
                                CHANGING SCR_0120-LANDX_FROM.
* 도착국가명을 가져온다.
    PERFORM SELECT_COUNTRY_NAME USING SPFLI-COUNTRYTO
                                CHANGING SCR_0120-LANDX_TO.

* 출발 공항명을 가져온다.
    PERFORM SELECT_AIRPORT_NAME USING SPFLI-AIRPFROM
                                CHANGING SCR_0120-AIRFROM_NAME.

* 도착 공항명을 가져온다.
    PERFORM SELECT_AIRPORT_NAME USING SPFLI-AIRPTO
                                CHANGING SCR_0120-AIRTO_NAME.


*----------------------------------------------------------------------*


  ELSE.
    GV_SELECT_RESULT = GC_SELECT_FAIL.

  ENDIF.

ENDMODULE.
