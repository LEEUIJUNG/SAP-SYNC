CLASS LCL_BUS IMPLEMENTATION.
  METHOD CONSTRUCTOR.
    IF IV_NUMBER IS INITIAL.
      RAISE EMPTY_NUMBER.
    ENDIF.

    IF IV_FARE = 0.
      RAISE FARE_ZERO.
    ENDIF.

    IF IV_FARE < 0.
      RAISE FARE_MINUS.
    ENDIF.

    MV_NUMBER = IV_NUMBER.
    MV_FARE = IV_FARE.

    GV_N_O_BUS += 1.

    " 현재 생성된 BUS 의 OBJ 를 ITAB 에 추가
    APPEND ME TO GT_B_BUS.

    WRITE:/ |버스 { MV_NUMBER } 이 생성되었습니다.|.

  ENDMETHOD.

  METHOD ADD_PASSENGER.
    MV_SALES += MV_FARE.
    MV_N_O_PASSENGER += 1.

    WRITE:/ |버스 { MV_NUMBER }에 승객이 탑승하여 매출이 증가합니다.|.


  ENDMETHOD.

  METHOD GET_FARE.
    " RV_FARE 를 할당 안하면 0 을 RETURN 한다 , (처음 RV_FARE이 0임으로 ).
    RV_FARE = MV_FARE.

  ENDMETHOD.

  METHOD DISPLAY_ATTRIBUTES.
    WRITE:/  '버스번호', AT 10 ':' , MV_NUMBER  ,
          / '운임', AT 10 ':' ,  MV_FARE ,'원',
          / '매출액', AT 10 ':' ,  MV_SALES  ,'원',
          / '승객수', AT 10 ':' ,  MV_N_O_PASSENGER  ,'원'.
    ULINE.

  ENDMETHOD.

  METHOD DISPLAY_N_O_BUS.
    WRITE:/ '생성된 총 버스 수 :', GV_N_O_BUS RIGHT-JUSTIFIED, '대'.
    ULINE.
  ENDMETHOD.

  METHOD DISPLAY_ALL_SALES.
    DATA: LV_TOTAL TYPE I.
    DATA: LO_BUS TYPE REF TO LCL_BUS.

    LOOP AT GT_B_BUS INTO LO_BUS.
      LV_TOTAL += LO_BUS->MV_SALES.
    ENDLOOP.

    WRITE:/ '생성된 총 매출액 :', LV_TOTAL RIGHT-JUSTIFIED, '원'.
    ULINE.
  ENDMETHOD.

  METHOD ARRIVE_TERMINAL.
    RAISE EVENT ARRIVED_TERMINAL.

  ENDMETHOD.

ENDCLASS.
