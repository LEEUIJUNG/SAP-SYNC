CLASS LCL_EVENT_HANDLER DEFINITION.
  " PUBLIC 으로 지정하여 외부에서 접근가능하도록 설정
  PUBLIC SECTION.
    " STATIC 으로 지정한다
    CLASS-METHODS: EVENT_TOOLBAR " TOOLBAR 관련 EVENT METHOD
      FOR EVENT TOOLBAR OF CL_GUI_ALV_GRID
      IMPORTING E_INTERACTIVE E_OBJECT.

    CLASS-METHODS: EVENT_USER_COMMAND " USER COMMAND 관련 EVENT METHOD
      FOR EVENT USER_COMMAND OF CL_GUI_ALV_GRID
      IMPORTING E_UCOMM.

    CLASS-METHODS: EVENT_DOUBLE_CLICK " DOBULE CLICK 관련 EVENT METHOD
      FOR EVENT DOUBLE_CLICK OF CL_GUI_ALV_GRID
      IMPORTING E_COLUMN E_ROW.


ENDCLASS.



CLASS LCL_EVENT_HANDLER IMPLEMENTATION.
  " TOOLBAR 관련 EVENT 선언
  METHOD EVENT_TOOLBAR.

    DATA: LS_TOOLBAR TYPE STB_BUTTON.     " TOOLBAR 를 저장하기 위한 STRUCT
    DATA: LV_ITAB_LINE_COUNT TYPE I.  " ITAB 이 몇줄인지

    LV_ITAB_LINE_COUNT = LINES( GT_PFLI ). " 현제 ITAB 이 몇줄인지 받아온다.

* 필터 조건 초기화 관련 TOOLBAR
    CLEAR LS_TOOLBAR.

    LS_TOOLBAR-FUNCTION = 'REST_EVENT_BUTTON'.
    LS_TOOLBAR-QUICKINFO = '모든 필터조건 초기화'.
    LS_TOOLBAR-TEXT =  |Total Lines: { LV_ITAB_LINE_COUNT }|.

    APPEND LS_TOOLBAR TO E_OBJECT->MT_TOOLBAR.

* 국내선 조회  관련 TOOLBAR
    CLEAR LS_TOOLBAR.

    LS_TOOLBAR-FUNCTION = 'DOMESTIC_AIRLINES_BUTTON'.
    LS_TOOLBAR-QUICKINFO = '국내선만 출력'.
    LS_TOOLBAR-TEXT =  '국내선'.

    APPEND LS_TOOLBAR TO E_OBJECT->MT_TOOLBAR.

* 국제선 조회 관련 TOOLBAR
    CLEAR LS_TOOLBAR.

    LS_TOOLBAR-FUNCTION = 'GLOBAL_AIRLINES_BUTTON'.
    LS_TOOLBAR-QUICKINFO = '국제선만 출력'.
    LS_TOOLBAR-TEXT =  '국제선'.

    APPEND LS_TOOLBAR TO E_OBJECT->MT_TOOLBAR.


  ENDMETHOD.

  " USER COMMAND 관련 EVENT 선언
  METHOD EVENT_USER_COMMAND.
* 현제는 수동으로 다시 ITAB 을 받아오지만
* FILTER 함수를 찾아서 추후 적용 예정

    DATA: LT_FILTER TYPE LVC_T_FILT. " FILTER 를 적용하기 위한 ITAB 선언.


    CASE E_UCOMM.
      WHEN 'REST_EVENT_BUTTON'. " 필터 초기화 버튼 클릭시
        REFRESH GT_FILTER
        .
        " FILTER 를 초기화 한다
        CALL METHOD GO_ALV_GRID->SET_FILTER_CRITERIA
          EXPORTING
            IT_FILTER = GT_FILTER.         " Filter Conditions


      WHEN 'GLOBAL_AIRLINES_BUTTON'. " 국제선 버튼 클릭시
        LT_FILTER = GT_FILTER. " 현재 적용중인 GLOBAL FILTER 를 가져온다.

        CLEAR GS_FILTER.

        GS_FILTER-FIELDNAME = 'ICON'.
        GS_FILTER-SIGN = 'I'.
        GS_FILTER-OPTION = 'EQ'.
        GS_FILTER-LOW = '@LS@'. " ICON_OUTGOING_OBJECT 의 값.

        APPEND GS_FILTER TO LT_FILTER.


        CALL METHOD GO_ALV_GRID->SET_FILTER_CRITERIA
          EXPORTING
            IT_FILTER = LT_FILTER.         " Filter Conditions



      WHEN 'DOMESTIC_AIRLINES_BUTTON'. " 국내선 버튼 클릭시
        LT_FILTER = GT_FILTER. " 현재 적용중인 GLOBAL FILTER 를 가져온다.

        CLEAR GS_FILTER.

        GS_FILTER-FIELDNAME = 'ICON'.
        GS_FILTER-SIGN = 'I'.
        GS_FILTER-OPTION = 'EQ'.
        GS_FILTER-LOW = '@LR@'. " ICON_INCOMING_OBJECT 의 값.

        APPEND GS_FILTER TO LT_FILTER.


        CALL METHOD GO_ALV_GRID->SET_FILTER_CRITERIA
          EXPORTING
            IT_FILTER = LT_FILTER.         " Filter Conditions


    ENDCASE.

    CALL METHOD GO_ALV_GRID->REFRESH_TABLE_DISPLAY( ). " ALV 를 REFRESH 하여 USER COMMAND에 대한 화면 적용
  ENDMETHOD.

  " DOUBLE CLICK 관련 EVENT 선언
  METHOD EVENT_DOUBLE_CLICK.
* DOUBLE CLICK EVENT 시 해당 열의 값을 받아온다.
    CLEAR GS_PFLI.
    READ TABLE GT_PFLI INTO GS_PFLI INDEX E_ROW-INDEX.

* DOUBLE CLICK 시 해당 COL 에 따른 프로그램 동작 로직
    CASE E_COLUMN.
      WHEN 'CARRID'  OR 'CARRNAME' .
        " 선택한 CELL 이 CARRID 나 CARRNAME 일 경우
        SET PARAMETER ID: 'CAR' FIELD GS_PFLI-CARRID. " PARMETER ID 가 CAR 이다
        " 검색 방법은 해당 T-CODE 에 들어가서 TECHNICAL INFO 에서 확인가능
        CALL TRANSACTION 'BC402MCAR' AND SKIP FIRST SCREEN. " .T-CODE 를 호출하고 첫 입력 화면을 스킵한다.

      WHEN OTHERS.
        " 선택한 CELL 이 위 두 가지가 아닐경우
        SUBMIT BC402_INS_FLIGHT_LIST_OSQL " EXCUTABLE PROGRAM 으로 이동
         WITH PA_CAR = GS_PFLI-CARRID " BC402_INS_FLIGHT_LIST_OSQL 의 PARAMETER 참조
         WITH SO_CON = GS_PFLI-CONNID " SO 로 CONNID 를 받음으로 BODY[] 를 통하여 전달.
         AND RETURN.

    ENDCASE.


  ENDMETHOD.


ENDCLASS.
