FORM SELECT_DATA .

  SELECT *
    FROM SPFLI
    INTO CORRESPONDING FIELDS OF TABLE GT_PFLI
    WHERE CARRID IN SO_CAR
      AND CONNID IN SO_CON.

  SELECT CARRID CARRNAME
    FROM SCARR
    INTO CORRESPONDING FIELDS OF TABLE GT_CARR
    WHERE CARRID IN SO_CAR.

* 검색성능을 위한 BINARY SEARCH 를 사용하기 위해 READ TAB 의 검색조건 순서대로 정렬 되어 있어야함.
  SORT GT_CARR BY CARRID.

ENDFORM.

FORM MAKE_DISPLAY_DATA .

  DATA: LS_FLTIME_YELLOW TYPE LVC_S_SCOL,
        LS_FLTIME_RED    TYPE LVC_S_SCOL.

  LS_FLTIME_YELLOW-FNAME = 'FLTIME'.
  LS_FLTIME_YELLOW-COLOR-COL = 3. " 노랑
  LS_FLTIME_YELLOW-COLOR-INT = 1. " 0: 강조 안함, 1 : 강조 진하게
  LS_FLTIME_YELLOW-COLOR-INV = 0. " 0 : 배경색, 1 : 글자색

  LS_FLTIME_RED-FNAME = 'FLTIME'.
  LS_FLTIME_RED-COLOR-COL = 6.
  LS_FLTIME_RED-COLOR-INT = 1.
  LS_FLTIME_RED-COLOR-INV = 0.

  REFRESH GT_DISPLAY.

  LOOP AT GT_PFLI INTO GS_PFLI.

    CLEAR GS_DISPLAY.

    " GS_PFLI가 가진 값들 중 동일한 이름을 지닌 필드끼리만 값 전달
    MOVE-CORRESPONDING GS_PFLI TO GS_DISPLAY.

    IF GS_DISPLAY-COUNTRYFR = GS_DISPLAY-COUNTRYTO.
      " 국내선 출발국가 도착 국가 동일
      GS_DISPLAY-ICON = ICON_INCOMING_OBJECT.

    ELSE.
      " 국제선 출발국가 도착 국가 동일하지 않음
      GS_DISPLAY-ICON = ICON_OUTGOING_OBJECT.

    ENDIF.


    " CELL 단위 색상처리
    IF GS_DISPLAY-FLTIME >= 1000.
      " 1000 이상일 경우 빨강
      APPEND LS_FLTIME_RED TO GS_DISPLAY-CELL_COLOR.

    ELSEIF GS_DISPLAY-FLTIME >= 600.
      " 1000이하 600 이상일 경우 노랑
      APPEND LS_FLTIME_YELLOW TO GS_DISPLAY-CELL_COLOR.

    ENDIF.

    " 항공사 명에 대한 설정
    READ TABLE GT_CARR INTO GS_CARR WITH KEY CARRID = GS_DISPLAY-CARRID
    BINARY SEARCH.

    IF SY-SUBRC = 0.
      GS_DISPLAY-CARRNAME = GS_CARR-CARRNAME.
    ELSE.
      GS_DISPLAY-CARRNAME = 'None'.

    ENDIF.


    " GS_DISPLAY가 가진 정보를 GT_DISPLAY의 마지막 줄에 추가
    APPEND GS_DISPLAY TO GT_DISPLAY.

  ENDLOOP.


ENDFORM.

FORM SET_ALV_FIELDCAT_0100 .
* FCAT 초기화
  REFRESH GT_FCAT.

* 아이콘 추가
  CLEAR GS_FCAT.
  GS_FCAT-FIELDNAME = 'ICON'. " ITAB 의 ICON FIELD 에 대한 내용
  GS_FCAT-ICON = 'X'. " ICON FIELD 로 취급
  GS_FCAT-COL_POS = 5.
  GS_FCAT-COLTEXT = '국내선/국제선'(F01).

  APPEND GS_FCAT TO GT_FCAT.

  CLEAR GS_FCAT.

  GS_FCAT-FIELDNAME = 'CARRNAME'. " ITAB 에 사용할 FIELD 명
  GS_FCAT-REF_FIELD = 'CARRNAME'. " DICT 의 FIELD 를 참조한다.
  GS_FCAT-REF_TABLE = 'SCARR'. " DICT를 참조한다
  GS_FCAT-COL_POS = 3.

  APPEND GS_FCAT TO GT_FCAT.

* 기존필드 - 출발국가
* 기존 ITAB FIELD 속성 변경
  CLEAR  GS_FCAT.

  GS_FCAT-FIELDNAME = 'COUNTRYFR'.
  GS_FCAT-COLTEXT = 'Dep. Ctr'(F02).

  APPEND GS_FCAT TO GT_FCAT.

* 기존필드 - 도착국가
* 기존 ITAB FIELD 속성 변경
  CLEAR  GS_FCAT.

  GS_FCAT-FIELDNAME = 'COUNTRYTO'.
  GS_FCAT-COLTEXT = 'Arr. Ctr'(F03).

  APPEND GS_FCAT TO GT_FCAT.

* 기존 ITAB FIELD 속성 숨김처리
  CLEAR  GS_FCAT.

  GS_FCAT-FIELDNAME = 'FLTYPE'.
  GS_FCAT-NO_OUT = ABAP_ON. " X 숨김처리 ( ABAP_ ON = 'X' )

  APPEND GS_FCAT TO GT_FCAT.

ENDFORM.
