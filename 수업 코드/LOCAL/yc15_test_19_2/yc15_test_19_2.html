  ENDIF.

ENDDO.

      LT_SPFLI TYPE TABLE OF SPFLI.

  EXPORTING
    TABLEname = LT_SPFLI
  TABLES
    DFIES_TAB = LT_FIELDS
  EXCEPTIONS
    OTHERS    = 1.
