FORM CHECK_SFLIGHT CHANGING CS_SDYN_CONN TYPE SDYN_CONN.

  DATA LS_SFLIGHT TYPE SFLIGHT.

  SELECT SINGLE * FROM SFLIGHT INTO LS_SFLIGHT
         WHERE CARRID = CS_SDYN_CONN-CARRID
         AND   CONNID = CS_SDYN_CONN-CONNID
         AND   FLDATE = CS_SDYN_CONN-FLDATE.
  IF SY-SUBRC = 0.
    MOVE-CORRESPONDING LS_SFLIGHT TO CS_SDYN_CONN.
  ELSE.
    MESSAGE E023 WITH CS_SDYN_CONN-CARRID
                      CS_SDYN_CONN-CONNID
                      CS_SDYN_CONN-FLDATE.
  ENDIF.

ENDFORM. " CHECK_SFLIGHT

*&---------------------------------------------------------------------*
*&      Form  READ_BOOKINGS
*&---------------------------------------------------------------------*
*      -->PV_CARRID  text
*      -->PV_CONNID  text
*      -->PV_FLDATE  text
*      <--CT_BOOK    text
*      <--CT_CD      text
*----------------------------------------------------------------------*
FORM READ_BOOKINGS USING PV_CARRID TYPE S_CARR_ID
                            PV_CONNID TYPE S_CONN_ID
                            PV_FLDATE TYPE S_DATE
                   CHANGING CT_BOOK   TYPE GTY_T_BOOKINGS
                            CT_CD     TYPE GTY_T_CD.

  DATA LS_CD TYPE SBOOK.
  FIELD-SYMBOLS <LS_BOOK> TYPE GTY_S_BOOKINGS. " 실제 변수를 만드는 것이 아니라
  " 기존에 존재하는 변수에게 새로운 이름을 부여한다.
  " STRUCT(gty_s_bookings) 를 가르킬 수 있는 변수 이름이다


  CLEAR CT_BOOK.
  CLEAR CT_CD.

  SELECT * FROM SBOOK INTO CORRESPONDING FIELDS OF TABLE CT_BOOK
         WHERE CARRID = PV_CARRID
         AND   CONNID = PV_CONNID
         AND   FLDATE = PV_FLDATE.

  LOOP AT CT_BOOK ASSIGNING <LS_BOOK>.
*   SCUSTOM is buffered (single dataset)
    SELECT SINGLE NAME FROM SCUSTOM INTO <LS_BOOK>-NAME
           WHERE ID = <LS_BOOK>-CUSTOMID.
    MOVE-CORRESPONDING <LS_BOOK> TO LS_CD.
    APPEND LS_CD TO CT_CD.
  ENDLOOP.
  SORT CT_BOOK BY BOOKID CUSTOMID.

ENDFORM. " READ_BOOKINGS

FORM MODIFY_BOOKINGS USING PT_BOOKINGS_MOD LIKE GT_BOOKINGS_MOD.

* 내가 취소 CHECK BOX 를 선택한 데이터만 기록된 PT_BOOKINGS_MOD의 내용을
* DB TAB (SBOOK) 에 저장
*  UPDATE SBOOK FROM TABLE PT_BOOKINGS_MOD.

  CALL FUNCTION 'UPDATE_SBOOK'
    EXPORTING
      IT_SBOOK       = PT_BOOKINGS_MOD               " Data sets to be updated in DB table SBOOK
    EXCEPTIONS
      UPDATE_FAILURE = 1                " Error during update of table SBOOK
      OTHERS         = 2.

  IF SY-SUBRC = 0.
    " 수정 성공
    " DB TAB SFLIGHT 에도 SBOOK의 변경 내용을 반영

    CALL FUNCTION 'UPDATE_SFLIGHT'
      EXPORTING
        IV_CARRID        = SDYN_CONN-CARRID               " Airline Carrier
        IV_CONNID        = SDYN_CONN-CONNID                " Connection Number
        IV_FLDATE        = SDYN_CONN-FLDATE           " Date
      EXCEPTIONS
        UPDATE_FAILURE   = 1                " Error during update of table SBOOK
        FLIGHT_FULL      = 2                " Flight is fully booked
        FLIGHT_NOT_FOUND = 3
        OTHERS           = 4.

    CASE SY-SUBRC.
      WHEN 0.
        MESSAGE S034 WITH SDYN_CONN-CARRID SDYN_CONN-CONNID SDYN_CONN-FLDATE.
      WHEN 1.
        MESSAGE A044 DISPLAY LIKE 'E' WITH SDYN_CONN-CARRID SDYN_CONN-CONNID SDYN_CONN-FLDATE.
        ROLLBACK WORK.

      WHEN 2.
        MESSAGE I045 DISPLAY LIKE 'E'.
        ROLLBACK WORK.

      WHEN 3.
        MESSAGE I046 DISPLAY LIKE 'E'.
        ROLLBACK WORK.

      WHEN OTHERS.
        MESSAGE I048 DISPLAY LIKE 'E'.
        ROLLBACK WORK.
    ENDCASE.

  ELSE.
    " 수정 실패 오류 메세지와 함꼐 ROLLBACK
    MESSAGE I004 WITH SDYN_CONN-CARRID SDYN_CONN-CONNID SDYN_CONN-FLDATE.

    ROLLBACK WORK.

  ENDIF.

ENDFORM.
