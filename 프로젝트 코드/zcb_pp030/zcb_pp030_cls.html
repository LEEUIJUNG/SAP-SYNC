  METHOD on_toolbar.
    DATA ls_button LIKE LINE OF e_object->mt_toolbar.

*    " ALV Grid 구분 - sender로 어떤 ALV 그리드인지 확인
    CASE sender.
      WHEN go_alv_gird.  " 첫 번째 ALV (상단 컨테이너)
        " 구분자
        CLEAR ls_button.
        ls_button-butn_type = 3.
        APPEND ls_button TO e_object->mt_toolbar.

        " 첫 번째 ALV 버튼 추가
        CLEAR ls_button.
        ls_button-function = 'TOOLBAR_SELECT_T001W'.
        ls_button-text     = '확정'. " 라우팅 해더 확정
        ls_button-icon     = '@0R@'.
        APPEND ls_button TO e_object->mt_toolbar.

        " 두 번째 ALV 버튼 추가
        CLEAR ls_button.
        ls_button-function = 'TOOLBAR_DELETE_T001W'.
        ls_button-text     = '삭제'. " 라우팅 해더 후보행 삭제
        ls_button-icon     = '@0R@'.
        APPEND ls_button TO e_object->mt_toolbar.

      WHEN go_alv_gird3.
        " 구분자
        CLEAR ls_button.
        ls_button-butn_type = 3.
        APPEND ls_button TO e_object->mt_toolbar.

        " 수정모드 / 수정중 툴바 버튼
        CLEAR ls_button.
        ls_button-function = 'TOOLBAR_SELECT_PLPO'.
        IF gv_plpo_edit_flag = 'X'.
          ls_button-text = '수정'. " 수정 중 상태
          ls_button-icon = '@0S@'.   " 예: 편집 중 아이콘
        ELSE.
          ls_button-text = '확정'. " 기본 상태
          ls_button-icon = '@0R@'.
        ENDIF.
        APPEND ls_button TO e_object->mt_toolbar.


    ENDCASE.
  ENDMETHOD.

  METHOD on_user_command.
    " ALV Grid 구분 - sender로 어떤 ALV 그리드인지 확인
    CASE sender.
      WHEN go_alv_gird.  " 첫 번째 ALV
        CASE e_ucomm.
            " Plant 헤더 확정 버튼
          WHEN 'TOOLBAR_SELECT_T001W'.

            IF gt_t001w IS INITIAL.
              MESSAGE 'Plant를 선택해 주세요.' TYPE 'S' DISPLAY LIKE 'E'.
              RETURN.
            ENDIF.

            IF gv_t001w_select_flag = ''.
              gv_t001w_select_flag = 'X'.

              " PLKO 입력 가능하도록 화면 재호출
              CALL SCREEN 0100.
            ELSE.
              MESSAGE '이미 Plant가 확정되었습니다.' TYPE 'S' DISPLAY LIKE 'E'.
            ENDIF.

            " Plant 해더 후보 행 삭제
          WHEN 'TOOLBAR_DELETE_T001W'.

            IF gv_t001w_select_flag <> ''.
              MESSAGE '이미 Plant가 확정되었습니다.' TYPE 'S' DISPLAY LIKE 'E'.
              RETURN.
            ENDIF.

            DATA: lt_index_rows TYPE lvc_t_row,
                  ls_index_row  TYPE lvc_s_row.

            CALL METHOD go_alv_gird->get_selected_rows(
              IMPORTING
                et_index_rows = lt_index_rows ).

            IF lt_index_rows IS INITIAL.
              MESSAGE '삭제할 행을 선택하세요.' TYPE 'S' DISPLAY LIKE 'E'.
              RETURN.
            ENDIF.

            " 역순 정렬 후 삭제
            SORT lt_index_rows BY index DESCENDING.

            LOOP AT lt_index_rows INTO ls_index_row.
              DELETE gt_t001w INDEX ls_index_row-index.
            ENDLOOP.

            PERFORM alv_refresh_0100.

        ENDCASE.

      WHEN go_alv_gird3.
        CASE e_ucomm.
          WHEN 'TOOLBAR_SELECT_PLPO'.

            IF gt_plpo IS NOT INITIAL.
              IF gv_plpo_edit_flag = ''.
                " PLPO ALV 변경 사항 ITAB에 적용
                PERFORM alv_itab_change USING go_alv_gird3.


                PERFORM set_alv_fcat_plpo_0100 USING ''.
                PERFORM rset_alv_fact_0100 USING gt_fcat3 go_alv_gird3.

                gv_plpo_edit_flag = 'X'.

                PERFORM alv_refresh_0100.

              ELSE.

                PERFORM set_alv_fcat_plpo_0100 USING 'X'.
                PERFORM rset_alv_fact_0100 USING gt_fcat3 go_alv_gird3.

                gv_plpo_edit_flag = ''.
                PERFORM alv_refresh_0100.

              ENDIF.

            ELSE.
              MESSAGE 'PLPO에 값이 없습니다.' TYPE 'S' DISPLAY LIKE 'E'.
            ENDIF.

        ENDCASE.
    ENDCASE.
  ENDMETHOD.
