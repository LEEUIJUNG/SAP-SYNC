*& Form get_crhd_table
*&---------------------------------------------------------------------*
FORM get_crhd_table USING i_meins TYPE zcb_plko-meins.
  " 라우팅 단위를 받아옴
  " 해당 단위를 사용하는 작업장만 가지고온다.
  " kg -> ea 단위 변환 방지 제품별 생산 공정은 한 단위로만 쭉 진행이 된다.

  " 기존 작업장 목록 초기화
  REFRESH gt_crhd.
  CLEAR gs_crhd.

  " 선택된 플랜트 수 계산
  DATA(lv_werks_count) = lines( gt_t001w ).

  " 플랜트가 1개 선택된 경우 해당 플랜트의 모든 작업장 가져오기
  IF lv_werks_count = 1.

    CLEAR gs_t001w.
    READ TABLE gt_t001w INTO gs_t001w INDEX 1.

    SELECT werks objid arbpl ltxa1 meng mein
      FROM zcb_crhd
      INTO CORRESPONDING FIELDS OF TABLE gt_crhd
     WHERE lvorm = ''
       AND werks = gs_t001w-werks
       AND mein = i_meins
     ORDER BY objid.

    " 플랜트가 여러 개 선택된 경우 공통 작업장(arbpl)만 추출
  ELSE.

    DATA: lt_common_arbpl TYPE TABLE OF zcb_crhd, " 공통 작업장 후보
          lt_temp_arbpl   TYPE TABLE OF zcb_crhd, " 현재 플랜트 작업장 목록
          lt_next_common  TYPE TABLE OF zcb_crhd, " 다음 공통 후보
          lv_first        TYPE c VALUE 'X'.       " 최초 플래그

    CLEAR gs_t001w.

    LOOP AT gt_t001w INTO gs_t001w.

      " 현재 플랜트의 작업장 목록 조회
      SELECT werks objid arbpl ltxa1 meng mein
        FROM zcb_crhd
        INTO CORRESPONDING FIELDS OF TABLE lt_temp_arbpl
       WHERE lvorm = ''
         AND werks = gs_t001w-werks
        AND mein = i_meins.

      " 최초 플랜트인 경우 → 기준 공통 목록 설정
      IF lv_first = 'X'.
        lt_common_arbpl = lt_temp_arbpl.
        lv_first = ''.
      ELSE.
        " 이후 플랜트들 → 공통 작업장만 추출
        CLEAR lt_next_common.

        LOOP AT lt_common_arbpl INTO DATA(ls_common).
          READ TABLE lt_temp_arbpl INTO DATA(ls_temp)
            WITH KEY arbpl = ls_common-arbpl.
          IF sy-subrc = 0.
            " arbpl이 공통으로 존재하는 경우만 유지
            APPEND ls_common TO lt_next_common.
          ENDIF.
        ENDLOOP.

        " 공통 작업장 목록 갱신
        lt_common_arbpl = lt_next_common.
      ENDIF.

    ENDLOOP.

    " 최종 공통 작업장 결과 설정
    gt_crhd = lt_common_arbpl.
    SORT gt_crhd BY arbpl.

  ENDIF.

ENDFORM.
