MODULE set_scr_0100 OUTPUT.
* 생산일자 - 오늘날짜, 플랜트 - 선택한 공장
  gv_input_plant = pa_werks.
  gv_input_date = sy-datum.

* 화면에 생산오더 상태 표시
  DATA: lv_red    TYPE i VALUE 0,
        lv_yellow TYPE i VALUE 0,
        lv_green  TYPE i VALUE 0.

  DATA: ls_afko LIKE LINE OF gt_afko.

  " 초기화 안시키면 메모리에 값이 남아 있음.
  lv_red = 0.
  lv_yellow = 0.
  lv_green = 0.


  LOOP AT gt_afko INTO ls_afko.
    IF ls_afko-icon = icon_led_red.
      lv_red += 1.
    ELSEIF ls_afko-icon = icon_led_yellow.
      lv_yellow += 1.
    ELSEIF ls_afko-icon = icon_led_green.
      lv_green += 1.
    ENDIF.
  ENDLOOP.

  gv_input_red = lv_red.
  gv_input_yellow = lv_yellow.
  gv_input_green = lv_green.

  " tick 순서
  gv_input_tick = gv_process_status.

  " 생산 스캐줄 생성 완료시
  IF gv_flag_sched = 'X'.
    CALL FUNCTION 'ICON_CREATE'
      EXPORTING
        name   = 'ICON_CHECKED'                " Icon name  (Name from INCLUDE <ICON> )
      IMPORTING
        result = gv_input_icon1.                 " Icon (enter the screen field here)
  ELSE.
    CALL FUNCTION 'ICON_CREATE'
      EXPORTING
        name   = 'ICON_CANCEL'                " Icon name  (Name from INCLUDE <ICON> )
      IMPORTING
        result = gv_input_icon1.                 " Icon (enter the screen field here)
  ENDIF.

  " 자재 입고 완료
  IF gv_flag_mard = 'X'.
    CALL FUNCTION 'ICON_CREATE'
      EXPORTING
        name   = 'ICON_CHECKED'                " Icon name  (Name from INCLUDE <ICON> )
      IMPORTING
        result = gv_input_icon2.                 " Icon (enter the screen field here)

  ELSE.
    CALL FUNCTION 'ICON_CREATE'
      EXPORTING
        name   = 'ICON_CANCEL'                " Icon name  (Name from INCLUDE <ICON> )
      IMPORTING
        result = gv_input_icon2.                 " Icon (enter the screen field here)

MODULE set_suba_0100 OUTPUT.
  CASE gv_tabstrip1-activetab.

  ENDCASE.
