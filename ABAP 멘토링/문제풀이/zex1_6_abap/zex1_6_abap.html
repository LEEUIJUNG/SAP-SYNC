REPORT ZEX1_6_ABAP.

SELECTION-SCREEN BEGIN OF BLOCK BLOCK1 WITH FRAME.
  PARAMETERS : GV_CAR TYPE SPFLI-CARRID DEFAULT 'AA',
               GV_CON   TYPE SPFLI-CONNID DEFAULT 17.
SELECTION-SCREEN END OF BLOCK BLOCK1.


DATA :GT_PFLI LIKE TABLE OF SPFLI.
DATA :GS_PFLI TYPE SPFLI.

SELECT *
  FROM SPFLI
  INTO CORRESPONDING FIELDS OF TABLE GT_PFLI
  WHERE CARRID = GV_CAR
  AND CONNID = GV_CON.

  WRITE : 'Schedule Information'.
  ULINE.
  LOOP AT GT_PFLI INTO GS_PFLI.


    WRITE: / GS_PFLI-CARRID, GS_PFLI-CONNID, GS_PFLI-COUNTRYFR, GS_PFLI-COUNTRYTO,
     GS_PFLI-CITYFROM,  GS_PFLI-CITYTO,  GS_PFLI-AIRPFROM,  GS_PFLI-AIRPTO.

  ENDLOOP.
